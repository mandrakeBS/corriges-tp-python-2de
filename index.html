6 
7def trouve_bornes(n, x0, delta=0.1): 
8    precision = 10 ** (-n) 
9    a = x0  # Point de départ. 
10    b = x0 + delta  # Point suivant. 
11    fa = f(a) 
12    fb = f(b) 
13    # Cherche les bornes d’encadrement. 
14    while abs(b - a) > precision: 
15        if fa * fb > 0:  # Pas de changement de signe. 
16            a = b 
17            b += delta 
18            fa = fb 
19            fb = f(b) 
20        else:  # Changement de signe trouvé. 
21            return a, b 
22    return a, b 
23 
24# Exécution du programme.
